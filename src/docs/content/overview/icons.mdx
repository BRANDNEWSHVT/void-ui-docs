export function IconGrid() {
  const [search, setSearch] = useState('')
  const [selectedCategory, setSelectedCategory] = useState('All')
  const [copiedIcon, setCopiedIcon] = useState(null)

  const categories = ['All', ...new Set(iconList.map(i => i.category))]

  const filteredIcons = iconList.filter(icon => {
    const matchesSearch = icon.name.toLowerCase().includes(search.toLowerCase())
    const matchesCategory = selectedCategory === 'All' || icon.category === selectedCategory
    return matchesSearch && matchesCategory
  })

  const handleCopy = async (iconName) => {
    await navigator.clipboard.writeText(`<${iconName} />`)
    setCopiedIcon(iconName)
    setTimeout(() => setCopiedIcon(null), 2000)
  }

  return (
    <div className="space-y-6">
      {/* Search and Filter */}
      <div className="flex flex-col gap-4">
        <Input
          placeholder="Search icons..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          className="max-w-xs"
        />
        <div className="flex flex-wrap gap-2">
          {categories.map(cat => (
            <button
              key={cat}
              onClick={() => setSelectedCategory(cat)}
              className={`rounded-full px-3 py-1 text-sm transition-colors ${selectedCategory === cat
                ? 'bg-(--void-primary) text-(--void-primary-foreground)'
                : 'bg-(--void-bg-subtle) text-(--void-muted) hover:bg-(--void-bg-muted)'
                }`}
            >
              {cat}
            </button>
          ))}
        </div>
      </div>

      {/* Icon Count */}
      <Text variant="muted" className="text-sm">
        Showing {filteredIcons.length} of {iconList.length} icons
      </Text>

      {/* Icon Grid */}
      <div className="grid grid-cols-4 gap-3 sm:grid-cols-6 md:grid-cols-8 lg:grid-cols-10">
        {filteredIcons.map(({ name }) => {
          const IconComponent = PhosphorIcons[name]
          if (!IconComponent) return null

          return (
            <button
              key={name}
              onClick={() => handleCopy(name)}
              className="group relative flex flex-col items-center gap-2 rounded-lg border border-(--void-border) bg-(--void-surface) p-3 transition-all hover:border-(--void-primary) hover:shadow-(--void-shadow-md)"
              title={name}
            >
              <IconComponent size={24} className="text-(--void-text)" />
              <span className="w-full truncate text-center text-xs text-(--void-muted)">
                {name}
              </span>
              {copiedIcon === name && (
                <span className="absolute -top-2 left-1/2 -translate-x-1/2 rounded bg-(--void-success) px-2 py-0.5 text-xs text-(--void-success-foreground)">
                  Copied!
                </span>
              )}
            </button>
          )
        })}
      </div>
    </div>
  )
}

# Icons

Void UI uses [Phosphor Icons](https://phosphoricons.com/) as the default icon library. Phosphor is a flexible icon family with 6 weights and over 7,000 icons.

## Installation

<CodeBlock
  code={`npm install @phosphor-icons/react`}
  language="bash"
/>

## API Reference

### Icon Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `number \| string` | `"1em"` | Icon size in pixels |
| `weight` | `"thin" \| "light" \| "regular" \| "bold" \| "fill" \| "duotone"` | `"regular"` | Icon weight/style |
| `color` | `string` | `"currentColor"` | Icon color |
| `mirrored` | `boolean` | `false` | Mirror the icon horizontally |
| `alt` | `string` | - | Accessible alt text |
