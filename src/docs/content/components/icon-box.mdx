# Icon Box

A container for displaying icons with consistent styling and semantic colors. Works great with Phosphor Icons.

## Preview

<ComponentPreview
  code={`import { IconBox } from '@radenadri/void-ui'
import { Lightning } from '@phosphor-icons/react'

export default function IconBoxDemo() {
  return (
    <Stack direction="row" spacing="md">
      <IconBox variant="primary">
        <Lightning weight="fill" />
      </IconBox>
      <IconBox variant="success-subtle">
        <CheckCircle weight="fill" />
      </IconBox>
      <IconBox variant="danger-subtle">
        <XCircle weight="fill" />
      </IconBox>
    </Stack>
  )
}`}
>
  <Stack direction="row" spacing="md">

      <CheckCircle weight="fill" />

    </IconBox>
  </Stack>
</ComponentPreview>
