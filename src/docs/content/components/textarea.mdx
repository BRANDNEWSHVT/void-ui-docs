### Controlled

<CodeBlock
  code={`import { useState } from 'react'

function ControlledTextarea() {
  const [value, setValue] = useState('')

  return (
    <Textarea 
      value={value} 
      onChange={(e) => setValue(e.target.value)}
      placeholder="Type something..."
    />
  )
}`}
  language="tsx"
/>
